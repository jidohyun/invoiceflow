## 4. 주요 산출물

### 4.1 ResetPasswordView + Universal Links

| 항목 | 내용 |
|---|---|
| 화면 | `ResetPasswordView(token: String)` |
| 진입 경로 | 이메일 링크 → Universal Link → `Scene.openURLContexts` |
| AASA | `apple-app-site-association` 의 `applinks.details[].components` 에 `/users/reset_password/*` |

`SceneDelegate`에서 URL 처리:
```swift
func scene(_ scene: UIScene, openURLContexts URLContexts: Set<UIOpenURLContext>) {
    guard let url = URLContexts.first?.url else { return }
    AppRouter.shared.handle(url)
}

func scene(_ scene: UIScene, continue userActivity: NSUserActivity) {
    guard userActivity.activityType == NSUserActivityTypeBrowsingWeb,
          let url = userActivity.webpageURL else { return }
    AppRouter.shared.handle(url)
}
```

`AppRouter`는 URL path를 매칭해 `/users/reset_password/{token}` 패턴이면 `ResetPasswordView(token:)`를 모달로 push.

### 4.3 BatchCaptureView (최대 10장)

| 항목 | 내용 |
|---|---|
| 카메라 | AVFoundation `AVCaptureSession` + photoOutput |
| 큐 모델 | `BatchCaptureViewModel.queue: [QueueItem]` (slot 최대 10) |
| 업로드 | `multipart/form-data` 동시 10 → 백엔드는 잡당 ExtractionJob 1건 |

상태:
```swift
struct BatchCaptureState: Equatable {
    var pending: [CapturedPhoto] = []
    var uploading: [String: UploadProgress] = [:]
    var completed: [CompletedExtraction] = []
    var failed: [FailedExtraction] = []
    var totalSlots = 10
    var remainingSlots: Int { max(0, totalSlots - pending.count - uploading.count) }
    var isFull: Bool { remainingSlots == 0 }
}
```

업로드 완료마다 `URLSessionWebSocketTask`로 백엔드 PubSub 결과를 받아 카드별 갱신. (백엔드가 LiveView PubSub → WebSocket 게이트웨이로 브리지)

## 5. 핵심 코드 발췌

### 5.1 ResetPasswordView — 토큰 검증 + 비밀번호 변경

```swift
@MainActor
final class ResetPasswordViewModel: ObservableObject {
    @Published var newPassword = ""
    @Published var confirm = ""
    @Published var state: State = .idle

    enum State: Equatable { case idle, submitting, success, failure(String) }

    private let api: AuthAPI

    init(token: String, api: AuthAPI = .live) {
        self.token = token; self.api = api
    }

    let token: String

    func submit() async {
        guard newPassword.count >= 8 else {
            state = .failure("비밀번호는 8자 이상이어야 합니다"); return
        }
        guard newPassword == confirm else {
            state = .failure("비밀번호가 일치하지 않습니다"); return
        }
        state = .submitting
        do {
            try await api.resetPassword(token: token, newPassword: newPassword)
            state = .success
        } catch {
            state = .failure("링크가 만료되었거나 잘못된 링크입니다.")
        }
    }
}
```

### 5.2 BatchCaptureViewModel — 동시 업로드 with TaskGroup

```swift
@MainActor
final class BatchCaptureViewModel: ObservableObject {
    @Published private(set) var state = BatchCaptureState()
    private let api: ExtractionAPI

    func submitAll() async {
        let toUpload = state.pending
        state.pending.removeAll()
        for photo in toUpload {
            state.uploading[photo.id] = .queued
        }

        await withTaskGroup(of: Void.self) { group in
            for photo in toUpload {
                group.addTask { [weak self] in
                    await self?.uploadOne(photo)
                }
            }
        }
    }

    private func uploadOne(_ photo: CapturedPhoto) async {
        do {
            for try await progress in api.uploadAndExtract(photo) {
                await MainActor.run { state.uploading[photo.id] = progress }
            }
            // PubSub 완료 이벤트 수신 시 completed로 이동
        } catch {
            await MainActor.run {
                state.uploading[photo.id] = nil
                state.failed.append(.init(id: photo.id, message: error.localizedDescription))
            }
        }
    }
}
```

### 5.3 ColorPickerSheet — SwiftUI 6.0 Color → HEX

```swift
struct ColorPickerSheet: View {
    @Binding var selection: Color
    let onCommit: (String) -> Void

    var body: some View {
        VStack {
            ColorPicker("브랜드 컬러", selection: $selection, supportsOpacity: false)
                .padding()
            Button("저장") {
                onCommit(selection.toHex())
            }
            .buttonStyle(.borderedProminent)
        }
    }
}

extension Color {
    func toHex() -> String {
        let comps = UIColor(self).cgColor.components ?? [0, 0, 0]
        return String(format: "#%02X%02X%02X",
                      Int(comps[0] * 255), Int(comps[1] * 255), Int(comps[2] * 255))
    }
}
```

## 6. 연구 내용 (신기술)

### 6.1 Universal Links + AASA — 안드로이드 App Links와 동등

**왜 연구했나:** 이메일 링크가 사파리로 열리면 모바일 사용 흐름이 끊긴다. iOS도 안드로이드 App Links와 같이 앱이 URL을 직접 받아야 한다.

**결론:** `Associated Domains` capability에 `applinks:auto-my-invoice.com` 추가 + 서버에 `.well-known/apple-app-site-association` 배포. iOS 시스템이 첫 실행 시 AASA를 가져와 검증, 이후 같은 패턴 URL은 시스템이 자동으로 앱으로 전달.

```json
{
  "applinks": {
    "details": [{
      "appIDs": ["TEAMID.com.automyinvoice.ios"],
      "components": [{ "/": "/users/reset_password/*" }]
    }]
  }
}
```

### 6.2 SwiftData → SwiftUI 6.0 Observation

**왜 연구했나:** 2차에서 SwiftData를 도입할 때 `@Published` + `ObservableObject` 패턴을 그대로 썼는데, SwiftUI 6.0의 `@Observable` 매크로가 성능·코드 양 모두 개선된다.

**결론:** ViewModel을 `@Observable` 매크로로 마이그레이션 시 boilerplate 30% 감소, 변경된 프로퍼티만 view re-render. BrandingSettingsViewModel을 시범 적용 → 4차에서 점진 마이그레이션.

```swift
@Observable
final class BrandingSettingsViewModel {
    var companyName = ""
    var brandColor: Color = .black
    // @Published 자동 처리, init 보일러플레이트 제거
}
```

### 6.3 TaskGroup + Streamed Progress

**왜 연구했나:** 다중 업로드 동안 카드별 진행률을 실시간으로 갱신하려면 동시 N개의 async 흐름이 서로 다른 stream을 메인 액터로 안전하게 보내야 한다.

**결론:** `withTaskGroup` 내부에서 각 업로드를 `AsyncThrowingStream<UploadProgress, Error>`로 표현하고, `for try await`로 받아 즉시 `await MainActor.run`으로 State 패치. 카드 10개 동시 업로드 시 메인 스레드 평균 부담 4% 이하 측정.
